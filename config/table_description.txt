# Items Table 
-> product_id varchar(45) Primary Key -- value between P01 to P10
-> product_name varchar(45) -- name of the product 
-> product_description varchar(100) -- description of the product 
-> product_price Double(10,2) -- price of the product 
-> emi_available varchar(10) -- Yes or No field 
-> vendor_id varchar(10) References users(user_id) -- id of the vendor suppling product 

# Users Table 
-> user_id varchar(10) -- value between 1-15 including users, admin, vendors
-> user_name varchar(45) -- name describing type of user(vendor, user, admin)
-> user_email varchar(45) -- email of the user 
-> user_password varchar(45) -- password of the user 
-> user_address varchar(45) -- address of the user all vendor address if Null 
-> is_vendor tinyint(1) -- boolean column indicating if user is vendor(1) or not(0)

# Orders Table 

-> Order_id int Primary Key -- index_value from orders table 
-> total_value Double(10,2) -- total order price as float 
-> order_qunatity INT -- quantity count 
-> reward_point INT -- reward_point awarded
-> vendor_id varchar(10) Foreign Key References Users(user_id) -- id of the vendor 
-> customer_id varchar(10) Foreign Key References Users(user_id) -- id of the customer
